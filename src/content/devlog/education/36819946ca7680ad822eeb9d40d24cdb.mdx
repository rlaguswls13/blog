<NotionIndent>


<NotionToggle title="Shape">

- **필드**
  - `type` : `int` → 도형 타입 번호 (예: 3 = 삼각형, 4 = 사각형)
  - `height` : `double` → 높이
  - `width` : `double` → 너비
  - `color` : `String` → 색상 (기본값 `"white"`)

- **생성자**
  - `Shape()` → 기본 생성자
  - `Shape(int type, double height, double width)` → 필드 초기화 생성자

- **메서드**

  - getter/setter 메서드 직접 구현
  - `information()` : `String` → `"높이 너비 색상"` 형식의 문자열 반환


</NotionToggle>


<NotionToggle title="SquareController">

- **필드**
  - `s` : `Shape` 객체 (기본 생성자로 생성)

- **메서드**

  - `calcPerimeter(double height, double width)` : `double`
    → `s`를 매개변수 있는 생성자로 초기화한 후, **둘레**(`width*2 + height*2`) 반환

  - `calcArea(double height, double width)` : `double`
    → `s`를 매개변수 있는 생성자로 초기화한 후, **넓이**(`width * height`) 반환

  - `paintColor(String color)` : `void`
    → `s`의 색상을 setter로 변경

  - `print()` : `String`

    → `"사각형 " + s.information()` 반환


</NotionToggle>

<NotionToggle title="ShapeMenu">

  - `printInformation(int type)` : `void`

```java
public class Shape {
    private int type;
    private double height;
    private double width;
    private String color = "white"; // 기본값 지정

    // 기본 생성자
    public Shape() {}

    // 필드 초기화 생성자
    public Shape(int type, double height, double width) {
        this.type = type;
        this.height = height;
        this.width = width;
    }

    // getter / setter 직접 구현
    public int getType() { return type; }
    public void setType(int type) { this.type = type; }

    public double getHeight() { return height; }
    public void setHeight(double height) { this.height = height; }

    public double getWidth() { return width; }
    public void setWidth(double width) { this.width = width; }

    public String getColor() { return color; }
    public void setColor(String color) { this.color = color; }

    // 정보 반환 메서드
    public String information() {
        return String.format("%.1f %.1f %s", height, width, color);
    }
}

}
```

```java
public class ShapeController {
    @Getter
    private Shape s; // 다형성을 활용하여 Triangle과 Square를 모두 수용

        }
    }

    public String print() {

    }
```

```java
import lombok.Getter;
import lombok.RequiredArgsConstructor;
import java.util.Arrays;

public class Menu {

    @Getter
    @RequiredArgsConstructor
    public enum MainMenu {
        TRIANGLE(1, "삼각형"),
        SQUARE(2, "사각형"),
        EXIT(9, "프로그램 종료"),
        NONE(0, "잘못된 메뉴");

        private final int code;
        private final String description;

        public static MainMenu fromCode(int code) {
            return Arrays.stream(values())
                    .filter(m -> m.code == code)
                    .findFirst()
                    .orElse(NONE);
        }
    }

    @Getter
    @RequiredArgsConstructor
    public enum TriangleMenu {
        CALC_AREA(1, "삼각형 면적 계산"),

        PRINT(3, "삼각형 정보 보기"),
        BACK(9, "메인으로"),
        NONE(0, "잘못된 메뉴");

        private final String description;

        public static TriangleMenu fromCode(int code) {
            return Arrays.stream(values())
                    .filter(m -> m.code == code)

                    .orElse(NONE);
        }
    }
```

<NotionToggle title="4. 메뉴 서비스 별도 구성 (상수, lamda식 Switch)">


```java
import java.util.Arrays;
import java.util.Scanner;

public class ShapeMenu {
    private final Scanner sc = new Scanner(System.in);
    private final ShapeController ctrl = new ShapeController();

    // UI 메시지 상수화로 하드코딩 완전 제거
    private static final String MSG_INPUT_SIZE = "높이 너비 입력 (공백 구분) : ";
    private static final String MSG_INPUT_COLOR = "색상 입력 : ";
    private static final String MSG_INVALID_MENU = "잘못된 번호입니다. 다시 입력하세요.";

    // 자동화된 공통 메뉴 출력 템플릿
    private <T extends Enum<T>> void renderMenu(String title, T[] menuValues) {
        System.out.println("\n===== " + title + " =====");
        Arrays.stream(menuValues)
                .filter(m -> !m.name().equals("NONE"))
                .forEach(m -> {
                    try {
                        int code = (int) m.getClass().getMethod("getCode").invoke(m);
                        String desc = (String) m.getClass().getMethod("getDescription").invoke(m);
                        System.out.printf("%d. %s\n", code, desc);
                    } catch (Exception ignored) {}
                });
        System.out.print("메뉴 번호 입력 : ");
    }

    // 반복되는 입력 행위 공통 메서드로 추출
    private double[] inputDimensions() {
        System.out.print(MSG_INPUT_SIZE);
        double height = sc.nextDouble();
        double width = sc.nextDouble();
        return new double[]{height, width};
    }

    private String inputColor() {
        System.out.print(MSG_INPUT_COLOR);
        return sc.next();
    }

    public void inputMenu() {
        while (true) {
            renderMenu("메인 메뉴", Menu.MainMenu.values());
            Menu.MainMenu menu = Menu.MainMenu.fromCode(sc.nextInt());

            switch (menu) {
                case TRIANGLE -> triangleMenu();
                case SQUARE   -> squareMenu();
                case EXIT     -> { System.out.println("프로그램을 종료합니다."); return; }
                default       -> System.out.println(MSG_INVALID_MENU);
            }
        }
    }

    private void triangleMenu() {
        while (true) {
            renderMenu("삼각형 메뉴", Menu.TriangleMenu.values());
            Menu.TriangleMenu menu = Menu.TriangleMenu.fromCode(sc.nextInt());

            switch (menu) {
                case BACK -> { return; }
                case CALC_AREA -> {
                    double[] size = inputDimensions();
                    double area = ctrl.createTriangle(size[0], size[1]);
                    System.out.println("면적 : " + area);
                }
                case PAINT -> ctrl.paintColor(inputColor());
                case PRINT -> System.out.println(ctrl.print());
                default -> System.out.println(MSG_INVALID_MENU);
            }
        }
    }

    private void squareMenu() {
        while (true) {
            renderMenu("사각형 메뉴", Menu.SquareMenu.values());
            Menu.SquareMenu menu = Menu.SquareMenu.fromCode(sc.nextInt());

            switch (menu) {
                case BACK -> { return; }
                case CALC_PERI, CALC_AREA -> {
                    double[] size = inputDimensions();
                    boolean isArea = (menu == Menu.SquareMenu.CALC_AREA);
                    double res = ctrl.createSquare(size[0], size[1], isArea);
                    System.out.println("결과 : " + res);
                }
                case PAINT -> ctrl.paintColor(inputColor());
                case PRINT -> System.out.println(ctrl.print());
                default -> System.out.println(MSG_INVALID_MENU);
            }
        }
    }
}
```

  - 각 필드별 **getter / setter** 직접 구현
  - `information():String` — 담긴 데이터를 한 줄 문자열로 반환

```java
import java.util.EnumMap;
import java.util.Map;
import java.util.Scanner;

public class SnackMenu {
    private final Scanner sc = new Scanner(System.in);
    private final SnackController scr = new SnackController();

    // UI 안내 메시지 상수화
    private static final String PROMPT_CONFIRM = "저장한 정보를 확인하시겠습니까?(y/n) : ";
    private static final String MSG_INVALID_INPUT = "잘못된 입력입니다. 메뉴를 종료합니다.";

    public void menu() {
        // 1. 별도 추출한 메서드를 통해 입력 데이터 Map을 깔끔하게 받아옴
        Map<SnackField, Object> inputData = readInputData();

        // 2. 컨트롤러 저장 로직 호출 및 결과 출력
        String saveResult = scr.saveData(inputData);
        System.out.println(saveResult);

        // 3. 후속 확인 질의
        System.out.print(PROMPT_CONFIRM);
        String answer = sc.next();

        switch (answer.toUpperCase()) {
            case "Y" -> System.out.println(scr.confirmData());
            case "N" -> { /* 복귀 */ }
            default  -> System.out.println(MSG_INVALID_INPUT);
        }
    }

    // 로직의 가독성을 위해 분리한 동적 입력 처리 메서드
    private Map<SnackField, Object> readInputData() {
        Map<SnackField, Object> inputData = new EnumMap<>(SnackField.class);

        for (SnackField field : SnackField.values()) {
            System.out.print(field.getPrompt());
            
            if (field.getType() == Integer.class) {
                inputData.put(field, sc.nextInt());
            } else {
                inputData.put(field, sc.next());
            }
        }
        return inputData;
    }
}
```

```java
@RequiredArgsConstructor
public enum MenuAction {
    CREATE("1", "스낵 추가 (C)"),
    READ("2", "스낵 조회 (R)"),
    UPDATE("3", "스낵 수정 (U)"),
    DELETE("4", "스낵 삭제 (D)"),

    private final String command;
    private final String description;

    public static final String TITLE_MAIN = "스낵 통합 관리 시스템";

    // 입력받은 문자열을 기반으로 MenuAction 객체 반환
    public static MenuAction fromCommand(String input) {
        return Arrays.stream(values())
                .filter(m -> m.command.equalsIgnoreCase(input.trim()))
                .findFirst()
                .orElse(NONE);

    public static String getMenuBoard() {
        return Arrays.stream(values())
                .filter(action -> !action.command.matches("EXIT|QUIT|NONE"))
                .map(action -> action.command + ". " + action.description)
                .collect(Collectors.joining(" | "));
    }
}
```

```java
import java.util.HashMap;
import java.util.Map;
import java.util.Optional;

public class SnackMenuService {
    private final Map<Long, Snack> repository = new HashMap<>();
    private long sequence = 0L; 

    private static final String MSG_SAVE_SUCCESS = "성공적으로 저장되었습니다. 부여된 고유ID는 %d입니다.";
    private static final String MSG_UPDATE_SUCCESS = "성공적으로 수정되었습니다.";
    private static final String MSG_DELETE_SUCCESS = "성공적으로 삭제되었습니다.";
    private static final String MSG_NOT_FOUND = "해당 ID의 스낵 정보를 찾을 수 없습니다.";
    private static final String MSG_EMPTY_REPOSITORY = "현재 저장된 스낵이 없습니다.";

    public String saveData(Map<SnackField, Object> dataMap) {
        long id = ++sequence;
        Snack snack = Snack.builder()
                .id(id)
                .kind((String) dataMap.get(SnackField.KIND))
                .name((String) dataMap.get(SnackField.NAME))
                .flavor((String) dataMap.get(SnackField.FLAVOR))
                .numOf((Integer) dataMap.get(SnackField.NUM_OF))
                .price((Integer) dataMap.get(SnackField.PRICE))
                .build();

        repository.put(id, snack);
        return String.format(MSG_SAVE_SUCCESS, id);
    }

    public String confirmData(long id) {
        return Optional.ofNullable(repository.get(id))
                .map(Snack::information)
                .orElse(MSG_NOT_FOUND);
    }

    public void printAllSnacks() {
        if (repository.isEmpty()) {
            System.out.println(MSG_EMPTY_REPOSITORY);
            return;
        }
        repository.values().forEach(snack -> System.out.println(snack.information()));
    }

    public String updateData(long id, Map<SnackField, Object> dataMap) {
        Snack target = repository.get(id);
        if (target == null) return MSG_NOT_FOUND;

        Snack updatedSnack = target.update(
                (String) dataMap.get(SnackField.KIND),
                (String) dataMap.get(SnackField.NAME),
                (String) dataMap.get(SnackField.FLAVOR),
                (Integer) dataMap.get(SnackField.NUM_OF),
                (Integer) dataMap.get(SnackField.PRICE)
        );

        repository.put(id, updatedSnack);
        return MSG_UPDATE_SUCCESS;
    }

    public String deleteData(long id) {
        if (!repository.containsKey(id)) return MSG_NOT_FOUND;
        repository.remove(id);
        return MSG_DELETE_SUCCESS;
    }
}
```

```java
import java.util.EnumMap;
import java.util.Map;
import java.util.Scanner;

public class SnackMenuController {
    private final Scanner sc = new Scanner(System.in);
    private final SnackMenuService ctrl = new SnackMenuService ();

    // 메뉴 제어부 내부 전용 안내 상수
    private static final String MSG_INVALID_MENU = "잘못된 번호입니다. 다시 입력하세요.";
    private static final String MSG_SYSTEM_OFF = "시스템을 완전히 종료합니다.";
    private static final String PROMPT_READ_ID = "조회할 스낵의 고유 ID를 입력하세요 : ";
    private static final String PROMPT_UPDATE_ID = "수정할 스낵의 고유 ID를 입력하세요 : ";
    private static final String PROMPT_DELETE_ID = "삭제할 스낵의 고유 ID를 입력하세요 : ";
    private static final String MSG_UPDATE_NOTICE = "[ 새로운 정보를 입력해 주세요 ]";

    public void startSystem() {
        boolean isRunning = true;

        while (isRunning) {
            // MenuAction 내부 상수를 활용하여 메뉴판 출력
            System.out.println("\n===== " + MenuAction.TITLE_MAIN + " =====");
            System.out.println(MenuAction.getMenuBoard());
            System.out.print(MenuAction.PROMPT_MENU_SELECTION);
            
            MenuAction action = MenuAction.fromCommand(sc.next());

            switch (action) {
                case EXIT, QUIT -> {
                    isRunning = false;
                    System.out.println(MSG_SYSTEM_OFF);
                }
                case CREATE    -> executeCreate();
                case READ      -> executeRead();
                case UPDATE    -> executeUpdate();
                case DELETE    -> executeDelete();
                case PRINT_ALL -> ctrl.printAllSnacks();
                default        -> System.out.println(MSG_INVALID_MENU);
            }
        }
    }

    private void executeCreate() {
        Map<SnackField, Object> inputData = readInputData();
        System.out.println(ctrl.saveData(inputData));
    }

    private void executeRead() {
        System.out.print(PROMPT_READ_ID);
        System.out.println(ctrl.confirmData(sc.nextLong()));
    }

    private void executeUpdate() {
        System.out.print(PROMPT_UPDATE_ID);
        long id = sc.nextLong();
        
        System.out.println(MSG_UPDATE_NOTICE);
        Map<SnackField, Object> updateData = readInputData();
        System.out.println(ctrl.updateData(id, updateData));
    }

    private void executeDelete() {
        System.out.print(PROMPT_DELETE_ID);
        System.out.println(ctrl.deleteData(sc.nextLong()));
    }

    private Map<SnackField, Object> readInputData() {
        Map<SnackField, Object> inputData = new EnumMap<>(SnackField.class);
        for (SnackField field : SnackField.values()) {
            System.out.print(field.getPrompt());
            inputData.put(field, field.getType() == Integer.class ? sc.nextInt() : sc.next());
        }
        return inputData;
    }
}
```
